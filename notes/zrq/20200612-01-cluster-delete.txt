# -----------------------------------------------------
# Create a container to work with.
#[user@desktop]

    source "${HOME:?}/aglais.env"

    podman run \
        --rm \
        --tty \
        --interactive \
        --hostname openstacker \
        --env "clouduser=${AGLAIS_USER:?}" \
        --env "cloudname=${AGLAIS_CLOUD:?}" \
        --volume "${HOME:?}/clouds.yaml:/tmp/clouds.yaml:ro,z" \
        atolmis/openstack-client \
        bash

# -----------------------------------------------------
# Link our clouds.cfg file.
# https://docs.openstack.org/python-openstackclient/pike/configuration/index.html#clouds-yaml
#[root@openstacker]

    mkdir -p "${HOME:?}/.config/openstack"
    pushd "${HOME:?}/.config/openstack"
        ln -sf /tmp/clouds.yaml
    popd

# -----------------------------------------------------
# Set the cluster name.
#[user@openstacker]

    clustername=Augustus

# -----------------------------------------------------
# List our clusters.
#[user@openstacker]

    openstack \
        --os-cloud "${cloudname:?}" \
        coe cluster list

    >   +--------------------------------------+----------+------------------+------------+--------------+-----------------+
    >   | uuid                                 | name     | keypair          | node_count | master_count | status          |
    >   +--------------------------------------+----------+------------------+------------+--------------+-----------------+

    >   +--------------------------------------+----------+------------------+------------+--------------+-----------------+


# -----------------------------------------------------
# Delete our cluster.
#[user@openstacker]

    openstack \
        --os-cloud "${cloudname:?}" \
        coe cluster list \
            --format json \
    | jq -r '.[0] | .uuid'

    >   f5632b7e-87bd-46d6-820f-a20f1b66b6c8

    clusterid=$(
        openstack \
            --os-cloud "${cloudname:?}" \
            coe cluster list \
                --format json \
        | jq -r '.[0] | .uuid'
        )

    openstack \
        --os-cloud "${cloudname:?}" \
        coe cluster delete \
            "${clusterid:?}"

    >   Request to delete cluster f5632b7e-87bd-46d6-820f-a20f1b66b6c8 has been accepted.


    openstack \
        --os-cloud "${cloudname:?}" \
