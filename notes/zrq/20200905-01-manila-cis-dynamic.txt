# Create a Pod that mounts the volume.

    cat > /tmp/harinabla-pod.yaml << EOF
kind: Pod
apiVersion: v1
metadata:
  name: harinabla-pod
  namespace: default
spec:
  volumes:
    - name: share-data
      persistentVolumeClaim:
        claimName: harinabla-claim
    - name: local-data
      emptyDir: {}
  containers:
    - name: harinabla-container
      image: 'fedora:latest'
      volumeMounts:
        - name: share-data
          mountPath: /share-data
        - name: local-data
          mountPath: /local-data
      command: ["/bin/sh"]
      args:
        - "-c"
        - >-
          while true; do
          date >> /share-data/\${HOSTNAME}.log;
          sleep 1;
          done
EOF
