apiVersion: v1
kind: PersistentVolume
metadata:
  name: adazoga-volume
  labels:
    name: adazoga-volume
spec:
  accessModes:
  - ReadWriteMany
  capacity:
    storage: 4Gi
  csi:
    driver: cephfs.manila.csi.openstack.org
    nodeStageSecretRef:
      name: os-trustee
      namespace: kube-system
    nodePublishSecretRef:
      name: os-trustee

    volumeAttributes:
