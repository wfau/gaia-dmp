# Create our checksum timing loop
#[user@master01]

    timestep()
        {
        echo "----"
        date --iso-8601=s
        time -p md5sum "${1}"
        sleep 20
        }

    pushd /data/gaia/edr3/GEDR3_11932_GAIASOURCE

        while true
        do
            for filename in $(ls -1)
            do
                timestep "${filename}" |& tee /tmp/timestep.log
            done
        done

    popd
