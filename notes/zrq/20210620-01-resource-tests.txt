# -----------------------------------------------------
# Create our create, delete and list functions.
#[user@openstacker]

    clouds=(
        gaia-dev
        gaia-test
        gaia-prod
        )

    flavorid()
        {
        local flavorname=${1:?}
        openstack \
            --os-cloud "${cloudname:?}" \
            flavor list \
                --format json \
            | jq -r '.[] | select(.Name == "'${flavorname}'") | .ID'
        }

    create-nodes()
        {
        local vmcount=${1:?}
        local flavorname=${2:?}
        local flavorid=$(flavorid "${flavorname}")

        echo ""
        echo "create-nodes"
        echo "  Flavor [${flavorname}]"

        for cloudname in ${clouds[*]}
        do
            echo "  Cloud  [${cloudname:?}]"

            for ((i=0 ; i < vmcount ; i++))
            do
                vmname=test-node-${i}
                echo "    Node   [${vmname}]"
                openstack \
                    --os-cloud "${cloudname}" \
                    server create \
                        --format json \
                        --image "${imageid:?}" \
                        --flavor "${flavorid:?}" \
                        --network "${networkid:?}" \
                        --key-name "${keyname:?}" \
                        "${vmname:?}" \
                | jq '.' \
                > '/tmp/${cloudname:?}-${vmname:?}.json'
            done
        done
        }

    list-all()
        {
        echo ""
        echo "list-all"

        for cloudname in ${clouds[*]}
        do
            echo "  Cloud [${cloudname:?}]"
            openstack \
                --os-cloud "${cloudname:?}" \
                server list
        done
        }

    list-active()
        {
        echo ""
        echo "list-active"

        for cloudname in ${clouds[*]}
        do
            echo "  Cloud [${cloudname:?}]"

            openstack \
                --os-cloud "${cloudname:?}" \
                server list \
                    --format json \
            | jq -r '.[] | select(.Status == "ACTIVE") | "\(.Flavor) \(.Name)"'
        done
        }

    list-faults()
        {
        echo ""
        echo "list-faults"

        for cloudname in ${clouds[*]}
        do
            echo "  Cloud [${cloudname:?}]"

            for serverid in $(
                openstack \
                    --os-cloud "${cloudname:?}" \
                    server list \
                        --format json \
                | jq -r '.[] | select(.Status != "ACTIVE") | .ID'
                )
            do
                echo "    Server [${serverid:?}]"
                message=$(
                    openstack \
                        --os-cloud "${cloudname:?}" \
                        server show \
                            --format json \
                            "${serverid:?}" \
                    | jq -r '.fault.message'
                    )
                echo "    Fault  [${message:?}]"
            done
        done
        }

    delete-all()
        {
        echo ""
        echo "delete-all"

        for cloudname in ${clouds[*]}
        do
            echo "  Cloud [${cloudname:?}]"
            for serverid in $(
                openstack \
                    --os-cloud "${cloudname:?}" \
                    server list \
                        --format json \
                | jq -r '.[] | .ID'
                )
            do
                echo "    Deleting [${serverid}]"
                openstack \
                    --os-cloud "${cloudname:?}" \
                    server delete \
                        "${serverid:?}"
            done
        done
        }

    create-test()
        {
        local vmcount=${1:?}
        local flavorname=${2:?}

        echo ""
        echo "create-test"
        echo "  Flavor [${flavorname}]"

        delete-all

        create-nodes "${vmcount:?}" "${flavorname:?}"

        list-active

        }
